// 반복문을 통해 데이터를 처리하는 예시 let currentIndex = 0; let jContent = ''; // J열에 저장될 텍스트 let kContent = ''; // K열에 저장될 텍스트 // H열: 텍스트를 계속 추가하고, https나 빈칸이 나오면 I열로 넘어감 while (currentIndex < columns.length && !columns[currentIndex].trim().startsWith('https') && columns[currentIndex].trim() !== '') { hContent += columns[currentIndex].trim() + ' '; currentIndex++; } // I열: https URL이 나오면 I열에 저장 while (currentIndex < columns.length && columns[currentIndex].trim().startsWith('https')) { iContent.push(columns[currentIndex].trim()); currentIndex++; } // J열: https가 포함된 텍스트는 J열에, 없으면 빈칸 if (currentIndex < columns.length && columns[currentIndex].trim().startsWith('https')) { jContent = columns[currentIndex].trim(); currentIndex++; // J열에 URL이 있을 경우 넘어가도록 설정 } else { jContent = ''; // URL이 없으면 빈칸으로 설정 } // K열: J열 이후 텍스트는 K열에 저장 while (currentIndex < columns.length && !columns[currentIndex].trim().startsWith('https') && columns[currentIndex].trim() !== '') { kContent += columns[currentIndex].trim() + ' '; currentIndex++; } // 리스트 항목 생성 let listItem = document.createElement('li'); // H열 텍스트 출력 listItem.innerHTML += `

H열: ${hContent.trim()}

`; // I열 URL 출력 iContent.forEach((url, index) => { listItem.innerHTML += `

이미지${index + 1} (I열): ${url}

`; }); // J열 텍스트 출력 listItem.innerHTML += `

J열: ${jContent.trim()}

`; // K열 텍스트 출력 listItem.innerHTML += `

이미지 (K열): ${kContent.trim()}

`; list.appendChild(listItem);
